  body{

.content_container:before,.content_container:after {

    position: absolute;
    top: 0;
    color: rgb(80, 240, 17);
    font-size: 40px;
    line-height: 40px;

    animation-name: opacity;
    animation-duration: 3s;
    animation-iteration-count: infinite;
}

.content_container_text{

    display: inline;

    margin: 0;

.content_container_list{

}
